# Calculate the first 10,000 terms of the series 1 - 1/3 + 1/5 - 1/7 + ...
total = 0
for i in range(10000):
    term = 1 / (2 * i + 1)  # Generate the odd denominators (1, 3, 5, 7, ...)
    if i % 2 == 0:  # Even index: add the term
        total += term
    else:  # Odd index: subtract the term
        total -= term

# Multiply the total by 4
result = total * 4

# Output the result
print(result)
